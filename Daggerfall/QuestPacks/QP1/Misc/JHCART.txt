_entry_ task:
    start timer _1st_
    give pc _letter_ notify 1010
    place npc _qg_ at _place_
    pc at _target_ set _inside_

_qgclicked_ task:
    clicked npc _qg_

_enroll_ task:
    when _qgclicked_ and not _begin_
    prompt 1012 yes _begin_ no _refuse_
    clear _qgclicked_

_refuse_ task:
    say 1015
    clear _qgclicked_ _refuse_ _enroll_

_begin_ task:
    say 1013
    log 1030 step 0
    stop timer _1st_
    start timer _timer_

_inside_ task:
    start task _found_

_found_ task:
    say 1020

_reminder_ task:
    when _qgclicked_ and _begin_ and not _found_
    say 1099
    clear _qgclicked_

_victory_ task:
    when _qgclicked_ and _found_
    start quest JHCGENT0
    give pc _reward_
    end quest

    end quest
