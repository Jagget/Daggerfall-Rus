Foe _enemy_ is Knight
Foe _q1_ is Wraith

    start timer _timer_
    get item _pearl_
    get item _bell_
    get item _amu_
    pick one of _e1_ _e2_ _e3_
    pc at _res_ do _inside_

_e1_ task:
    log 1030 step 0

_e2_ task:
    log 1031 step 0

_e3_ task:
    log 1032 step 0

_inside_ task:
    say 1010

_act1_ task:
    when _e1_ and _inside_
    _amu_ used do _trapa_
    _pearl_ used do _trapb_
    _bell_ used do _wina_

_act2_ task:
    when _e2_ and _inside_
    _bell_ used do _trapc_
    _amu_ used do _trapd_
    _pearl_ used do _winb_

_act3_ task:
    when _e3_ and _inside_
    _pearl_ used do _trape_
    _bell_ used do _trapf_
    _amu_ used do _winc_

_infight_ task:
    daily from 0:00 to 23:59
    change foe _q1_ infighting true
    change foe _enemy_ infighting true
    clear _infight_

_trapa_ task:
    say 1011
    teleport pc to _dung_
    make pc ill with Plague

_trapb_ task:
    say 1012
    create foe _q1_ every 0 minutes 6 times with 100% success

_wina_ task:
    say 1013
    create foe _enemy_ every 0 minutes 1 times with 100% success

_trapc_ task:
    say 1015
    teleport pc to _dung_
    make pc ill with Plague

_trapd_ task:
    say 1016
    create foe _q1_ every 0 minutes 6 times with 100% success

_winb_ task:
    say 1017
    create foe _enemy_ every 0 minutes 1 times with 100% success

_trape_ task:
    say 1019
    teleport pc to _dung_
    make pc ill with Plague

_trapf_ task:
    say 1020
    create foe _q1_ every 0 minutes 6 times with 100% success

_winc_ task:
    say 1021
    create foe _enemy_ every 0 minutes 1 times with 100% success

_slain_ task:
    killed 1 _enemy_
    say 1023

_qgclicked_ task:
    clicked npc _qgiver_

_clearclick_ task:
    when _qgclicked_ and not _slain_
    clear _qgclicked_ _clearclick_

_pcreward_ task:
    when _qgclicked_ and _slain_
    give pc nothing
    end quest

_timer_ task:
    end quest
