    start timer _timer_
    pc at _dung_ set _inside_
    log 1030 step 0
    reveal _dung_
    pc at any fighters set _fginside_
    place npc _kid_ at _dung_

_inside_ task:
    say 1017

variable _fginside_

_kidclick_ task:
    clicked npc _kid_
    say 1010
    hide npc _kid_
    add _kid_ face
    start task _prompt_

_prompt_ task:
    prompt 1011 yes _train_ no _ignore_

_ignore_ task:
    say 1014

_skilled_ task:
    when skill LongBlade is at least 50

variable _train_

_yestrain_ task:
    when _skilled_ and _train_
    say 1012
    start task _ready_

variable _ready_

_notrain_ task:
    when not _skilled_ and _train_
    say 1013
    start task _badweapon_

variable _badweapon_

_leavedungeon_ task:
    when _kidclick_ and not _inside_
    say 1015
    start task _ambush_

variable _ambush_

_easyend_ task:
    when _ambush_ and _ready_
    say 1016

_hardend_ task:
    when _ambush_ and not _ready_
    create foe _asn_ every 0 minutes 1 times with 100% success

_badreward_ task:
    when _qgclicked_ and _kidclick_ and _badweapon_
    say 1019
    give pc _smallreward_
    end quest

_pcgetsgold_ task:
    when _qgclicked_ and _kidclick_ and not _badweapon_
    start quest JHFGC05
    say 1018
    give pc _reward_
    end quest

_qgclicked_ task:
    clicked npc _questgiver_

_clearclick_ task:
    when _qgclicked_ and not _kidclick_
    clear _qgclicked_ _clearclick_

    end quest
