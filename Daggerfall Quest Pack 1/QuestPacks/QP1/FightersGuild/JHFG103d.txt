_enter_ task:
    when pc enters city
    clear _noenter_

_noenter_ task:
    when pc exits city
    clear _enter_

_spawn_ task:
    when _enter_ and _delay_
    start task _dospawn_

_dospawn_ task:
    say 1010
    create foe _enemy_ every 0 minutes 3 times with 100% success
    start timer _ending_

_infight_ task:
    daily from 0:00 to 23:59
    change foe _enemy_ infighting true
    clear _infight_

_ending_ task:
    end quest
