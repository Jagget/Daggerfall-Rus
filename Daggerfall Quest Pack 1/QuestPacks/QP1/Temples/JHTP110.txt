Clock _timer_ 60.00:00

--	Quest start-up:
    log 1030 step 0
    start timer _timer_
    place npc _person_ at _res_
    pc at _farhouse_ do _inside_

_talkpers_ task:
    clicked npc _person_
    say 1010
    add _person_ face
    hide npc _person_
    place npc _kid_ at _farhouse_
    log 1031 step 1

_talkkid_ task:
    clicked npc _kid_
    prompt 1011 yes _help_ no _deny_
    drop _person_ face
    hide npc _kid_
    stop timer _timer_

_help_ task:
    say 1012
    place foe _kidwolf_ at _farhouse_
    legal repute -10
    change foe _kidwolf_ infighting true

_kill_ task:
    killed _kidwolf_
    say 1013
    give pc nothing
    legal repute -20
    end quest

_deny_ task:
    say 1014
    place foe _spellsword_ at _farhouse_
    change foe _spellsword_ infighting true

_killman_ task:
    killed _spellsword_
    say 1015
    give pc nothing
    end quest

variable _inside_
_coward_ task:
    when _deny_ and not _inside_
    say 1016
    end quest

_escape_ task:
    when _help_ and not _inside_
    say 1018
    give pc nothing
    end quest

_endtalk_ task:
    when _timer_ and _talkpers_
    say 1017

_timer_ task:
    end quest
