QBN:

Place _magery_ local magery

Item _item_ magic_item

Clock _timer_ 12:00

--	Quest start-up:
    prompt 1011 yes _pay_ no _reject_

_reject_ task:
    say 1012
    end quest

variable _pay_

_payhim_ task:
    when _pay_ and not _given_
    pay 20 money do _given_ otherwise do _nomoney_

_nomoney_ task:
    say 1014
    end quest

_given_ task:
    pick one of _tip_ _end_ _end_ _end_ _end_

_end_ task:
    say 1013
    end quest

_daytime_ task:
    daily from 11:00 to 23:00

_tip_ task:
    say 1015
    log 1010 step 0

_plant_ task:
    when _tip_ and not _daytime_
    place item _item_ at _magery_
    start timer _timer_
    have _item_ set _grab_

_grab_ task:
    make _item_ permanent
    end quest

_timer_ task:
    end quest
