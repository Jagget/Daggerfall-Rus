    end quest

_int70_ task:

_int50_ task:

_bigwin_ task:
    when _given_ and _int90_
    say 1012
    give pc _reward1_
    end quest

_medwin_ task:
    when _given_ and _int70_ and not _int90_
    say 1013
    give pc _reward2_
    end quest

_smallwin_ task:
    when _given_ and _int50_ and not _int70_
    say 1014
    give pc _reward3_
    end quest

_fail_ task:
    when _given_ and not _int50_
    say 1015
    end quest
